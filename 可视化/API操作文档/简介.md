
```javascript
const app = new Application({
  container: document.getElementById("container"),
});

const rect = app.create({
  type: "Rect",
  width: 100,
  height: 100,
  x: app.Viewer.BackgroundRect.width() / 2,
  y: app.Viewer.BackgroundRect.height() / 2,
});
rect.Background = "#4583ff";

app.Viewer.zoomAll();
```
